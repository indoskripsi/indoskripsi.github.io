= (8/(√5 + √13)) × ((√5 - √13)/(√5 - √13))

= (8(√5 - √13))/(5 - 13)

= (8(√5 - √13))/(5 - 13)

= (8(√5 - √13))/(-8)

= -(√5 - √13)

= √13 - √5
